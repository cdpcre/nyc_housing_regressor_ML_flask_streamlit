üíª Code Examples

import requests

# Single prediction
response = requests.post('{{ request.url_root }}predict', json={
    "brokertitle": "Brokered by COMPASS",
    "type": "Condo for sale",
    "beds": 2,
    "bath": 1.0,
    "propertysqft": 800.0,
    "sublocality": "Manhattan"
})

result = response.json()
print(f"Predicted price: {result['price_formatted']}")

# Health check
health = requests.get('{{ request.url_root }}health')
print(f"API Status: {health.json()['status']}")

// Single prediction
fetch('{{ request.url_root }}predict', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json',
    },
    body: JSON.stringify({
        brokertitle: "Brokered by COMPASS",
        type: "Condo for sale",
        beds: 2,
        bath: 1.0,
        propertysqft: 800.0,
        sublocality: "Manhattan"
    })
})
.then(response => response.json())
.then(data => {
    console.log('Predicted price:', data.price_formatted);
})
.catch(error => console.error('Error:', error));

# Single prediction
curl -X POST {{ request.url_root }}predict \
  -H "Content-Type: application/json" \
  -d '{
    "brokertitle": "Brokered by COMPASS",
    "type": "Condo for sale",
    "beds": 2,
    "bath": 1.0,
    "propertysqft": 800.0,
    "sublocality": "Manhattan"
  }'

# Health check
curl {{ request.url_root }}health
